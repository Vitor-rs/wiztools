/* ===== utilitários ===== */
function sh(nome){
var s = SpreadsheetApp.getActive().getSheetByName(nome);
if (!s) throw new Error('Aba "'+nome+'" não existe neste arquivo — o painel precisa dela (copie da planilha-mãe ou recrie).');
return s;
}
function lerTabela(nome){
var v = sh(nome).getDataRange().getValues();
return { cab: v[0], linhas: v.slice(1).filter(function(r){ return r.join('') !== ''; }) };
}
function hhmm(v){
if (v instanceof Date) return Utilities.formatDate(v, SpreadsheetApp.getActive().getSpreadsheetTimeZone(), 'HH:mm');
if (typeof v === 'number'){ var m = Math.round(v*24*60); return ('0'+Math.floor(m/60)).slice(-2)+':'+('0'+(m%60)).slice(-2); }
return String(v||'');
}
function statusDe(situacao){ // dirigido pela aba Lista_Situacoes (coluna Ativa 1/0)
var s = getSituacoes().filter(function(x){ return x.situacao===situacao; })[0];
return (s && s.ativa) ? 'Ativado' : 'Desativado';
}
function getSituacoes(){ // domínio: aba Lista_Situacoes (auto-criada com os padrões se não existir; depois a tabela manda)
var ss = SpreadsheetApp.getActive(), s = ss.getSheetByName('Lista_Situacoes');
if (!s){
s = ss.insertSheet('Lista_Situacoes');
s.getRange(1,1,7,2).setValues([['Situacao','Ativa'],['Matriculado',1],['Rematriculado',1],['Cancelado',0],['Encerrado',0],['Trancado',0],['Evadido',0]]);
s.getRange(1,1,1,2).setFontWeight('bold');
}
var v = s.getDataRange().getValues();
return v.slice(1).filter(function(r){ return String(r[0])!==''; }).map(function(r){ return { situacao:String(r[0]), ativa:Number(r[1])===1 }; });
}
function getModalidades(){ // domínio derivado de tb_Prioridade_Aula_Blocos (base sem o prefixo 'Vip ')
var out = [];
lerTabela(ABAS.PRIORIDADE).linhas.forEach(function(r){
var m = String(r[0]).replace(/^Vip\s+/i,'');
if (m && out.indexOf(m)<0) out.push(m);
});
return out;
}

/* ===== domínios (lidos pelo painel na abertura) ===== */
function getDominios(){
var cfg = lerTabela(ABAS.CONFIG);
var nd = 0; while (nd+1 < cfg.cab.length && String(cfg.cab[nd+1]) !== '') nd++;
var dias = cfg.cab.slice(1, nd+1);
var horariosPorDia = {};
dias.forEach(function(d){ horariosPorDia[d]=[]; });
cfg.linhas.forEach(function(r){
var hora = hhmm(r[0]);
dias.forEach(function(d,i){ if (Number(r[i+1])===1) horariosPorDia[d].push(hora); });
});
var lv = lerTabela(ABAS.LIVROS);
var livros = lv.linhas.map(function(r){ return { nome:String(r[0]), tipoPadrao:r[2]||'Conn', kids:r[3]===true, tipoFixo:r[4]===true }; });
var fun = lerTabela(ABAS.FUNC);
var professores = fun.linhas.map(function(r){ return { id:r[0], nomeCompleto:r[1], nome:r[2] }; });
return {
situacoes: getSituacoes().map(function(x){ return x.situacao; }),
modalidades: getModalidades(),
dias: dias.filter(function(d){ return horariosPorDia[d].length>0; }),
horariosPorDia: horariosPorDia,
livros: livros,
professores: professores,
turmas: getTurmas()
};
}

/* ===== matriz de horários (edição via painel) ===== */
function nDiasCfg(v){ var n=0; while (n+1 < v[0].length && String(v[0][n+1]) !== '') n++; return n; } // pára na 1ª coluna vazia (ignora legenda)
function getMatriz(){
var v = sh(ABAS.CONFIG).getDataRange().getValues();
var n = nDiasCfg(v);
return { dias: v[0].slice(1, n+1),
horas: v.slice(1).map(function(r){ return hhmm(r[0]); }),
valores: v.slice(1).map(function(r){ return r.slice(1, n+1).map(function(x){ return Number(x)===1?1:0; }); }) };
}
function salvarMatriz(valores){ // grade [hora][dia] com 1/0 — só as colunas de dias
var s = sh(ABAS.CONFIG), v = s.getDataRange().getValues();
var n = nDiasCfg(v);
if (!valores || valores.length !== v.length-1 || valores[0].length !== n)
throw new Error('Dimensões da matriz não conferem com Config_Horarios.');
var out = valores.map(function(r){ return r.slice(0, n).map(function(x){ return Number(x)===1?1:0; }); });
s.getRange(2,2,out.length,n).setValues(out);
return { ok:true };
}

/* ===== ALUNOS (CRUD — fonte da verdade: Lista_Nominal_Config) ===== */
function getAlunos(){
return lerTabela(ABAS.NOMINAL).linhas.map(function(r){
return { nome:r[0], id:r[1], situacao:r[2], status:r[3] };
});
}
function salvarAluno(a){ // {id, nome, situacao} — cria ou atualiza (upsert por ID)
if (!a || !a.nome || String(a.nome).trim()==='' || String(a.id||'').trim()==='') throw new Error('Nome e ID de matrícula são obrigatórios.');
if (getSituacoes().every(function(x){ return x.situacao!==a.situacao; })) throw new Error('Situação inválida: '+a.situacao);
var s = sh(ABAS.NOMINAL), v = s.getDataRange().getValues(), linha = -1;
for (var i=1;i<v.length;i++){ if (String(v[i][1])===String(a.id)){ linha=i+1; break; } }
var registro = [String(a.nome).trim(), a.id, a.situacao, statusDe(a.situacao)];
if (linha>0) s.getRange(linha,1,1,4).setValues([registro]);
else s.getRange(v.length+1,1,1,4).setValues([registro]);
var nAulas = getAulasAluno(a.id).length;
return { ok:true, criado:linha<0, status:statusDe(a.situacao), avisoAulas:(statusDe(a.situacao)==='Desativado' && nAulas>0) ? nAulas : 0 };
}
function excluirAluno(id){ // remove aluno + aulas em cascata
var s = sh(ABAS.NOMINAL), v = s.getDataRange().getValues();
for (var i=v.length-1;i>=1;i--){ if (String(v[i][1])===String(id)) s.deleteRow(i+1); }
var sa = sh(ABAS.AULAS), va = sa.getDataRange().getValues(), n=0;
for (var j=va.length-1;j>=1;j--){ if (String(va[j][0])===String(id)){ sa.deleteRow(j+1); n++; } }
return { ok:true, aulasRemovidas:n };
}

/* ===== AULAS (tabela combinatória: 1 linha = aluno × dia × horário) ===== */
function getAulasAluno(id){
var v = sh(ABAS.AULAS).getDataRange().getValues(), out=[];
for (var i=1;i<v.length;i++){
var r=v[i];
if (String(r[0])===String(id)) out.push({ linha:i+1, dia:r[2], horario:hhmm(r[3]), livro:r[4],
professores:String(r[5]||'').split(',').map(function(p){return p.trim();}).filter(String) });
}
return out;
}
function salvarAula(a){ // {idMatricula, dia, horario, livro, professores:[], linha?}
var dom = getDominios();
if (!dom.horariosPorDia[a.dia] || dom.horariosPorDia[a.dia].indexOf(a.horario)<0)
throw new Error('Horário '+a.horario+' não está ativado para '+a.dia+' (veja Config_Horarios).');
if (!dom.livros.some(function(l){ return l.nome===a.livro; })) throw new Error('Livro inválido: '+a.livro);
var aluno = getAlunos().filter(function(x){ return String(x.id)===String(a.idMatricula); })[0];
if (!aluno) throw new Error('Aluno '+a.idMatricula+' não encontrado.');
var profs = (a.professores||[]).join(', ');
var idFunc = '';
if ((a.professores||[]).length===1){
var f = dom.professores.filter(function(p){ return p.nome===a.professores[0]; })[0];
idFunc = f ? f.id : '';
}
var registro = [aluno.id, aluno.nome, a.dia, a.horario, a.livro, profs, idFunc];
var s = sh(ABAS.AULAS);
if (a.linha) s.getRange(a.linha,1,1,7).setValues([registro]);
else s.getRange(s.getLastRow()+1,1,1,7).setValues([registro]);
return { ok:true, turma: deduzirTurma(a.livro, a.dia, a.horario) };
}
function salvarAgendaLivro(p){ // {idMatricula, livro, professores:[], itens:[{dia,horario}]}
// Sincroniza a agenda do aluno NESTE livro: após salvar, as linhas dele neste livro são exatamente os dias marcados.
// Desmarcar = remover (não há exclusão manual). Também assume o lugar de aula de OUTRO livro no mesmo dia+horário
// (troca de livro do aluno). Aulas de outros livros em outros horários (ex.: 2º idioma) não são tocadas.
if (!p || !p.itens) throw new Error('Dados incompletos.');
var avisos = [];
var dom = getDominios();
if (!dom.livros.some(function(l){ return l.nome===p.livro; })) throw new Error('Livro inválido: '+p.livro);
p.itens.forEach(function(it){
if (!dom.horariosPorDia[it.dia] || dom.horariosPorDia[it.dia].indexOf(it.horario)<0)
throw new Error('Horário '+it.horario+' não está ativado para '+it.dia+' (veja a matriz).');
var tu = deduzirTurma(null, it.dia, it.horario, p.professores||[]);
if (tu && tu.livro!==p.livro)
avisos.push(it.dia+' '+it.horario+': o aluno fica na turma '+tu.nome+' com livro diferente ('+p.livro+' × '+tu.livro+') — considere atualizar o livro da turma.');
});
var s = sh(ABAS.AULAS), v = s.getDataRange().getValues(), removidas = 0;
for (var i=v.length-1;i>=1;i--){
var r=v[i];
if (String(r[0])!==String(p.idMatricula)) continue;
var slotNovo = p.itens.some(function(it){ return it.dia===String(r[2]) && it.horario===hhmm(r[3]); });
if (String(r[4])===p.livro || slotNovo){ s.deleteRow(i+1); removidas++; }
}
var n=0;
p.itens.forEach(function(it){
salvarAula({ idMatricula:p.idMatricula, dia:it.dia, horario:it.horario, livro:p.livro, professores:p.professores });
n++;
});
return { ok:true, salvas:n, removidas:removidas, avisos:avisos };
}

function deduzirTurma(livro, dia, horario, profs){ // casa Livro+Dia+Horário; professora desempata turmas gêmeas
var m = getTurmas().filter(function(t){
return t.status==='Ativa' && (!livro || !t.livro || t.livro===livro) && t.horario===horario && String(t.blocoDias).indexOf(dia)>=0;
});
if (m.length>1 && profs && profs.length){
var pm = m.filter(function(t){ return t.professores.some(function(x){ return profs.indexOf(x)>=0; }); });
if (pm.length) m = pm;
}
return m.length ? m[0] : null;
}
function nomeTurma(mod, vip, livro, dias, hIni, hFim, profs){
var p = (profs||[]).length ? ' | '+(profs||[]).join('/') : '';
return (vip?'VIP-':'Tur-')+String(mod).toUpperCase()+(livro?' | '+livro:'')+' | '+dias+' | '+hIni+'-'+hFim+p;
}
function atualizarLivroTurma(idTurma, novoLivro){ // troca o livro da turma e de TODAS as aulas dos alunos dela
var s = sh(ABAS.TURMAS), v = s.getDataRange().getValues();
var linha=-1, r0=null;
for (var i=1;i<v.length;i++){ if (String(v[i][0])===String(idTurma)){ linha=i+1; r0=v[i]; break; } }
if (linha<0) throw new Error('Turma '+idTurma+' não encontrada.');
var lv = lerTabela(ABAS.LIVROS).linhas.filter(function(x){ return String(x[0])===String(novoLivro); })[0];
if (!lv) throw new Error('Livro inválido: '+novoLivro);
var antigo = String(r0[2]), mod = r0[3], vip = r0[4]===true, dias = String(r0[5]), hini = hhmm(r0[6]), hfim = hhmm(r0[7]);
var profsT = String(r0[8]||'').split(',').map(function(x){return x.trim();}).filter(String);
s.getRange(linha,3).setValue(novoLivro);
s.getRange(linha,2).setValue(nomeTurma(mod,vip,novoLivro,dias,hini,hfim,profsT));
var sa = sh(ABAS.AULAS), va = sa.getDataRange().getValues(), nAulas=0;
for (var j=1;j<va.length;j++){
var r=va[j];
var okProf = !profsT.length || String(r[5]||'').split(',').some(function(x){ return profsT.indexOf(x.trim())>=0; });
if (String(r[4])===antigo && hhmm(r[3])===hini && dias.indexOf(String(r[2]))>=0 && okProf){ sa.getRange(j+1,5).setValue(novoLivro); nAulas++; }
}
return { ok:true, turma:idTurma, de:antigo, para:novoLivro, aulasAtualizadas:nAulas };
}
function salvarTurma(t){ // {id?, livro, modalidade?, vip, blocoDias, horario, horaFim, professores:[], status?}
if (!t.blocoDias || !t.horario || !t.horaFim) throw new Error('Dias e horários são obrigatórios.');
if (String(t.horaFim) <= String(t.horario)) throw new Error('Horário impossível: fim ('+t.horaFim+') deve ser depois do início ('+t.horario+').');
var dom = getDominios();
var lv = null;
if (t.livro){ lv = dom.livros.filter(function(l){ return l.nome===t.livro; })[0]; if (!lv) throw new Error('Livro inválido: '+t.livro); }
var mod = t.modalidade || (lv ? lv.tipoPadrao : 'Inter');
if (lv && lv.tipoFixo) mod = lv.tipoPadrao;
if (!t.livro && mod!=='Inter') throw new Error('Livro é obrigatório para turma '+mod+' — só turma Interactive é multi-livro (sem livro fixo).'); // TOTS / L. Kids: sempre Conn, nunca Interactive
if (getModalidades().indexOf(mod)<0) throw new Error('Modalidade inválida: '+mod);
var vip = t.vip===true;
var s = sh(ABAS.TURMAS), v = s.getDataRange().getValues();
var id = t.id, linha = -1;
var antigo = null;
if (id){ for (var i=1;i<v.length;i++){ if (String(v[i][0])===String(id)){ linha=i+1; antigo=v[i]; break; } } }
else { var max=0; for (var j=1;j<v.length;j++){ var n=parseInt(String(v[j][0]).replace(/\D/g,''),10)||0; if (n>max) max=n; } id='T'+('00'+(max+1)).slice(-3); }
var gemea = getTurmas().filter(function(x){
return String(x.id)!==String(id) && x.status==='Ativa' && x.livro===t.livro && x.horario===t.horario && String(x.blocoDias)===String(t.blocoDias);
}).filter(function(x){
var meus = t.professores||[];
if (!meus.length || !x.professores.length) return true; // sem professora não há como distinguir
return x.professores.some(function(p){ return meus.indexOf(p)>=0; });
})[0];
if (gemea) throw new Error('Já existe a turma '+gemea.id+' ('+gemea.nome+') com o mesmo livro, dias e horário. Vincule professoras DIFERENTES para diferenciar as duas turmas (salas distintas).');
var registro = [id, nomeTurma(mod,vip,t.livro,t.blocoDias,t.horario,t.horaFim,t.professores||[]), t.livro, mod, vip,
t.blocoDias, t.horario, t.horaFim, (t.professores||[]).join(', '), t.status||'Ativa', (vip?'Vip ':'')+mod];
if (linha>0) s.getRange(linha,1,1,11).setValues([registro]);
else s.getRange(v.length+1,1,1,11).setValues([registro]);
var aulasAtualizadas = 0;
if (antigo && String(antigo[2]) !== String(t.livro)){ // livro da turma mudou: atualiza as aulas de todos os alunos dela
var sa = sh(ABAS.AULAS), va = sa.getDataRange().getValues();
var diasAnt = String(antigo[5]), hAnt = hhmm(antigo[6]);
var profsAnt = String(antigo[8]||'').split(',').map(function(x){return x.trim();}).filter(String);
for (var k=1;k<va.length;k++){
var ra=va[k];
var okP = !profsAnt.length || String(ra[5]||'').split(',').some(function(x){ return profsAnt.indexOf(x.trim())>=0; });
if (String(ra[4])===String(antigo[2]) && hhmm(ra[3])===hAnt && diasAnt.indexOf(String(ra[2]))>=0 && okP){ sa.getRange(k+1,5).setValue(t.livro); aulasAtualizadas++; }
}
}
return { ok:true, id:id, nome:registro[1], aulasAtualizadas:aulasAtualizadas };
}

function getIntegrantesTurma(idTurma){
var t = getTurmas().filter(function(x){ return String(x.id)===String(idTurma); })[0];
if (!t) throw new Error('Turma '+idTurma+' não encontrada.');
var v = sh(ABAS.AULAS).getDataRange().getValues();
var por = {};
for (var i=1;i<v.length;i++){
var r=v[i], dia=String(r[2]), hora=hhmm(r[3]), livro=String(r[4]);
if (hora!==t.horario || String(t.blocoDias).indexOf(dia)<0) continue;
var profs = String(r[5]||'').split(',').map(function(x){ return x.trim(); }).filter(String);
var mesmaProf = profs.some(function(x){ return t.professores.indexOf(x)>=0; });
if (t.livro && livro!==t.livro && !mesmaProf) continue; // outro livro com OUTRA professora = aluno avulso no horário, não é da turma
if (t.professores.length && profs.length && !mesmaProf) continue; // turma gêmea: professora diferente = outra turma
var k = String(r[0]);
if (!por[k]) por[k] = { id:k, nome:String(r[1]), dias:[], livros:[] };
if (por[k].dias.indexOf(dia)<0) por[k].dias.push(dia);
if (por[k].livros.indexOf(livro)<0) por[k].livros.push(livro);
}
return Object.keys(por).map(function(k){
var m = por[k];
m.divergente = t.livro ? m.livros.some(function(x){ return x!==t.livro; }) : false; // muitos-para-muitos: permitido, só alerta
return m;
});
}
function removerAlunoDaTurma(idMatricula, idTurma){
// Remove o VÍNCULO (as aulas do aluno no dia/horário da turma). O cadastro do aluno permanece.
var t = getTurmas().filter(function(x){ return String(x.id)===String(idTurma); })[0];

function gerTitulo(b, d){ // título da coluna B (sem livro; VIP leva dias; Kids destacado)
var dias = b.turma ? String(b.turma.blocoDias).split('+') : [];
var curtos = dias.map(function(x){ return (d.dias[x.trim()]||{}).curto || x.trim(); }).join(' | ');
if (b.tipoKey==='Kids') return 'Turma Conn KIDS';
if (b.vip) return b.mod+' Vip'+(curtos?' - '+curtos:'');
if (b.mod==='Inter') return 'Turma Inter';
return 'Turma '+b.mod+(curtos?' - '+curtos:'');
}

function gerBloco(s, r, b, d){
var cores = GER.CORES[b.tipoKey] || GER.CORES['Conn'];
var nomes = Object.keys(b.alunos);
// header 1
s.getRange(r,1).setValue('Hora: '+b.hora+' - '+b.fim).setBackground(cores.a)
.setFontFamily('Calibri').setFontSize(13).setFontWeight('bold').setHorizontalAlignment('left').setVerticalAlignment('middle');
s.getRange(r,2,1,16).setBackground(cores.bq); // cor de B até Q; E:P ficam VAZIAS de propósito (dia da semana escrito à mão)
s.getRange(r,2,1,3).merge().setValue(gerTitulo(b, d))
.setFontFamily('Calibri').setFontSize(12).setFontWeight('bold').setHorizontalAlignment('center').setVerticalAlignment('middle');
s.getRange(r,17).setValue('Observações')
.setFontFamily('Calibri').setFontSize(12).setFontWeight('bold').setHorizontalAlignment('center').setVerticalAlignment('middle');
s.setRowHeight(r, GER.ALT.h1);
// header 2
var h2 = [['Aluno(a)','Dias','Livro','Aula','','','','','','','','','','','','','Professores']];
s.getRange(r+1,1,1,17).setValues(h2).setBackground(cores.sub)
.setFontFamily('Calibri').setFontSize(11).setFontWeight('bold').setHorizontalAlignment('center').setVerticalAlignment('middle');
s.setRowHeight(r+1, GER.ALT.h2);
// corpo (alunos) + linhas extras
var corpo = nomes.map(function(id){
var a = b.alunos[id];
var cods = a.dias.map(function(x){ return d.dias[x]||{cod:x,ordem:9}; }).sort(function(p,q){ return p.ordem-q.ordem; }).map(function(x){ return x.cod; }).join('|');
var tipoCel = b.vip ? (b.mod+' Vip') : (b.tipoKey==='Kids' ? 'Conn' : b.mod);
var row = ['','','','','','','','','','','','','','','','',''];
row[0]=a.nome; row[1]=cods; row[2]=a.livro; row[3]=tipoCel; row[16]=a.profs.join(', ')||b.profs.join(', ');
return row;
});
var n = corpo.length;
if (n){
s.getRange(r+2,1,n,17).setValues(corpo).setFontFamily('Calibri').setFontSize(11)
.setHorizontalAlignment('center').setVerticalAlignment('middle');
s.getRange(r+2,1,n,1).setHorizontalAlignment('left');
}
var total = 2 + n + GER.EXTRAS_POR_BLOCO;
for (var i=0;i<n+GER.EXTRAS_POR_BLOCO;i++) s.setRowHeight(r+2+i, GER.ALT.corpo);
s.getRange(r,1,total,17).setBorder(true,true,true,true,true,true);
return r + total;
}
