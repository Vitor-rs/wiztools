SCRIPT 1: Sincronizar
8
Fonte da verdade = tb_Lista_Nominal. No modelo v2, Nome e ID_Funcionario em tb_Aulas são COLUNAS CALCULADAS (fórmula), então o script não sincroniza mais nomes. (1) Ordena Nominal por nome. (2) Remove linhas de tb_Aulas cujo ID_Matricula não existe na Nominal. (3) Adiciona alunos Matriculado/Rematriculado sem nenhuma linha (1 linha mínima). (4) Reaplica as fórmulas das colunas calculadas (auto-cura).
9
function main(workbook: ExcelScript.Workbook) {
// ===== SCRIPT 1: SINCRONIZAR (v2 — modelo derivado) =====
// Fonte da verdade = tb_Lista_Nominal. Nome e ID_Funcionario em tb_Aulas sao COLUNAS CALCULADAS.
const MATRIC = ["Matriculado", "Rematriculado"];
const F_NOME = '=IF([@[ID_Matricula]]="","",XLOOKUP([@[ID_Matricula]],tb_Lista_Nominal[ID_Matricula],tb_Lista_Nominal[Nome],""))';
const F_FUNC = '=IF([@Professor]="","",XLOOKUP([@Professor],tb_Lista_Funcionarios[Nome],tb_Lista_Funcionarios[ID_Funcionarios],""))';

const tbNom = workbook.getTable("tb_Lista_Nominal");
const tbAulas = workbook.getTable("tb_Aulas");
if (!tbNom || !tbAulas) { console.log("ERRO: tb_Lista_Nominal ou tb_Aulas nao encontrada."); return; }

// --- 1) Ordena Nominal por Nome ---
tbNom.getSort().apply([{ key: 0, ascending: true }]);

// --- Mapa ID -> Status a partir da Nominal ---
const nomVals = tbNom.getRangeBetweenHeaderAndTotal().getValues();
const idToStatus: { [id: string]: string } = {};
for (const r of nomVals) {
const id = String(r[1] || "").trim();
if (id) idToStatus[id] = String(r[3] || "").trim();
}

// --- 2) Remove linhas cujo ID nao existe na Nominal (de baixo pra cima) ---
const aVals = tbAulas.getRangeBetweenHeaderAndTotal().getValues();
let fantasmas = 0;
for (let i = aVals.length - 1; i >= 0; i--) {
const id = String(aVals[i][0] || "").trim();
if (!id || idToStatus[id] === undefined) { tbAulas.deleteRowsAt(i, 1); fantasmas++; }
}

// --- 3) Adiciona matriculados sem nenhuma linha ---
const idsComAula: { [id: string]: boolean } = {};
for (const r of tbAulas.getRangeBetweenHeaderAndTotal().getValues()) {
const id = String(r[0] || "").trim();
if (id) idsComAula[id] = true;
}
let adicionados = 0;
for (const id of Object.keys(idToStatus)) {
if (MATRIC.indexOf(idToStatus[id]) < 0 || idsComAula[id]) continue;
tbAulas.addRow(-1, [id, "", "", "", "", "", "", "", ""]);
adicionados++;
}

// --- 4) Reaplica colunas calculadas (auto-cura: addRow sobrescreve formula com "") ---
tbAulas.getColumn("Nome").getRangeBetweenHeaderAndTotal().setFormula(F_NOME);
tbAulas.getColumn("ID_Funcionario").getRangeBetweenHeaderAndTotal().setFormula(F_FUNC);

console.log("SINCRONIZACAO CONCLUIDA");
console.log(" - Fantasmas removidos: " + fantasmas);
console.log(" - Alunos novos adicionados (sem aula): " + adicionados);
}

SCRIPT 2: Gerar Blocos
12
Agrupa por HORÁRIO → sub-blocos por MODALIDADE (prioridade tb_Prioridade_Aula_Blocos) → por PROFESSOR. Professores diferentes = sub-blocos separados, em ordem alfabética. VIP = 1 sub-bloco por aluno. Coluna Q = Professor (nome curto via Lista_Funcionarios, vinculado pelo ID_Funcionario na Lista_Aulas). Colunas A=Aluno B+C(mescl)=tipo no header / B=Dias C=Livro D=Aula, E:P presença, Q=Professor. Livro e Professor em fonte 16. Sem zebra, anti-corte 32 linhas, A4 paisagem.
13
function main(workbook: ExcelScript.Workbook) {
// ===== SCRIPT 2: GERAR BLOCOS (template fiel: horário + modalidade + professor) =====
const FIRST_DATA_ROW = 5; // blocos comecam na linha 5; linhas 1-4 = cabecalho de pagina
const ROWS_PER_PAGE = 32;
const GAP = 3;
const ROW_H = 14;
const MATRIC = ["Matriculado", "Rematriculado"];
const COR_HORA = "#90B5D4";
const COR_HEADER = "#A6C9EC";
const FONTE = "Calibri";

const ABAS: { sheet: string; dias: string[]; titulo: string }[] = [
{ sheet: "Blocos_Segundas_Quartas", dias: ["Segunda", "Quarta"], titulo: "SEGUNDAS e QUARTAS" },
{ sheet: "Blocos_Tercas_Quintas", dias: ["Terça", "Quinta"], titulo: "TERÇAS e QUINTAS" },
{ sheet: "Blocos_Sextas", dias: ["Sexta"], titulo: "SEXTAS" },
{ sheet: "Blocos_Sabados", dias: ["Sábado"], titulo: "SÁBADOS" },
];
const MESES = ["JANEIRO","FEVEREIRO","MARÇO","ABRIL","MAIO","JUNHO","JULHO","AGOSTO","SETEMBRO","OUTUBRO","NOVEMBRO","DEZEMBRO"];
const mesAtual = MESES[new Date().getMonth()];

function norm(s: string): string {
return String(s == null ? "" : s).normalize("NFD").replace(/[\u0300-\u036f]/g, "").toLowerCase().replace(/\s+/g, " ").trim();
}
function normHora(h: string | number): string {
if (typeof h === "number") { const t = Math.round(h * 24 * 60); const hh = Math.floor(t / 60); const mm = t % 60; return `${String(hh).padStart(2,"0")}:${String(mm).padStart(2,"0")}`; }
const m = String(h).trim().match(/(\d{1,2}):(\d{2})/); return m ? `${m[1].padStart(2,"0")}:${m[2]}` : String(h).trim();
}

// --- Prioridade modalidade ---
const prioByTipo: { [k: string]: number } = {};
const tbPri = workbook.getTable("tb_Prioridade_Aula_Blocos");
if (tbPri) { for (const r of tbPri.getRangeBetweenHeaderAndTotal().getValues()) { const t = norm(String(r[0]||"")); if (t) prioByTipo[t] = Number(r[1]) || 0; } }
function prioridade(tipo: string): number { const p = prioByTipo[norm(tipo)]; return p === undefined ? 999 : p; }
function ehVip(tipo: string): boolean { return norm(tipo).indexOf("vip") >= 0; }

// --- Professores: ID -> nome curto ---
const profById: { [id: string]: string } = {};
const tbFunc = workbook.getTable("tb_Lista_Funcionarios");
if (tbFunc) { for (const r of tbFunc.getRangeBetweenHeaderAndTotal().getValues()) { const id = String(r[0]||"").trim(); if (id) profById[id] = String(r[2]||"").trim(); } }

// --- Status por ID ---
const tbNom = workbook.getTable("tb_Lista_Nominal");
const statusById: { [id: string]: string } = {};
for (const r of tbNom.getRangeBetweenHeaderAndTotal().getValues()) { const id = String(r[1]||"").trim(); if (id) statusById[id] = String(r[3]||"").trim(); }

// --- Ordens ---
const horOrdem: { [k: string]: number } = {};
for (const r of workbook.getTable("tb_Lista_Horarios").getRangeBetweenHeaderAndTotal().getValues()) { const d = String(r[1]||"").trim(); if (d) horOrdem[normHora(d)] = Number(r[2]); }
const livOrdem: { [k: string]: number } = {};
for (const r of workbook.getTable("tb_Lista_Livros").getRangeBetweenHeaderAndTotal().getValues()) { const d = String(r[0]||"").trim(); if (d) livOrdem[d] = Number(r[1]) || 9999; }
const diaCod: { [k: string]: string } = {}; const diaOrd: { [k: string]: number } = {};
for (const r of workbook.getTable("tb_Lista_Dias").getRangeBetweenHeaderAndTotal().getValues()) { const d = String(r[0]||"").trim(); if (d) { diaCod[d]=String(r[1]||"").trim(); diaOrd[d]=Number(r[2])||99; } }

// --- Aulas validas: [0ID 1Nome 2Dia 3Hora 4Livro 5Tipo 6IDTurma 7Professor 8IDFunc] ---
const tbAulas = workbook.getTable("tb_Aulas");
const aulas = tbAulas.getRangeBetweenHeaderAndTotal().getValues();
const validas = aulas.filter(r => {
const id = String(r[0]||"").trim(); const dia = String(r[2]||"").trim(); const hora = String(r[3]||"").trim();
if (!id || !dia || !hora) return false;
return MATRIC.indexOf(statusById[id] || "") >= 0;
});

const resumo: string[] = [];

for (const aba of ABAS) {
const ws = workbook.getWorksheet(aba.sheet);
if (!ws) continue;
const aulasAba = validas.filter(r => aba.dias.indexOf(String(r[2]).trim()) >= 0);

const porHora: { [hora: string]: (string|number|boolean)[][] } = {};
for (const r of aulasAba) { const k = normHora(r[3] as string); (porHora[k] = porHora[k] || []).push(r); }
const horarios = Object.keys(porHora).sort((a,b) => (horOrdem[a]||999) - (horOrdem[b]||999));

type Aluno = { nome: string; dias: string[]; livro: string; tipo: string; prof: string };
type SubBloco = { hora: string; tipo: string; prof: string; prio: number; alunos: Aluno[] };
const subBlocos: SubBloco[] = [];

for (const hora of horarios) {
const porAluno: { [key: string]: Aluno } = {};
for (const r of porHora[hora]) {
const id = String(r[0]).trim();
const livro = String(r[4]||"").trim();
const tipo = String(r[5]||"").trim();
const idFunc = String(r[8]||"").trim();
const prof = profById[idFunc] || String(r[7]||"").trim();
const key = id + "|" + livro + "|" + tipo + "|" + prof;
if (!porAluno[key]) porAluno[key] = { nome: String(r[1]).trim(), dias: [], livro, tipo, prof };
const d = String(r[2]).trim();
if (porAluno[key].dias.indexOf(d) < 0) porAluno[key].dias.push(d);
}
const lista = Object.keys(porAluno).map(k => porAluno[k]);
// agrupa por tipo + professor
const grp: { [k: string]: Aluno[] } = {};
for (const a of lista) { const gk = a.tipo + "||" + a.prof; (grp[gk] = grp[gk] || []).push(a); }
for (const gk of Object.keys(grp)) {
const arr = grp[gk];
arr.sort((x,y) => (livOrdem[x.livro]||9999)-(livOrdem[y.livro]||9999) || x.nome.localeCompare(y.nome));
const tipo = arr[0].tipo; const prof = arr[0].prof;
if (ehVip(tipo)) { for (const a of arr) subBlocos.push({ hora, tipo, prof: a.prof, prio: prioridade(tipo), alunos: [a] }); }
else { subBlocos.push({ hora, tipo, prof, prio: prioridade(tipo), alunos: arr }); }
}
}

subBlocos.sort((a,b) => {
const ha = horOrdem[a.hora]||999, hb = horOrdem[b.hora]||999;
if (ha !== hb) return ha - hb;
if (a.prio !== b.prio) return a.prio - b.prio;
if (a.prof !== b.prof) return a.prof.localeCompare(b.prof);
return (a.alunos[0]?a.alunos[0].nome:"").localeCompare(b.alunos[0]?b.alunos[0].nome:"");
});

// Limpa tudo (inclui topo, vamos reescrever)
const used = ws.getUsedRange();
if (used) { const last = used.getLastRow().getRowIndex()+1; ws.getRange(`A1:Q${Math.max(last+30, 40)}`).clear(ExcelScript.ClearApplyTo.all); }

// ===== Cabecalho de pagina (linhas 1-4) =====
const tit = ws.getRange("A2:E3"); tit.merge(false);
tit.setValue(aba.titulo);
tit.getFormat().getFont().setName(FONTE); tit.getFormat().getFont().setSize(18); tit.getFormat().getFont().setBold(true);
tit.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.left);
tit.getFormat().setVerticalAlignment(ExcelScript.VerticalAlignment.center);
const mes = ws.getRange("L2:P3"); mes.merge(false);
mes.setValue(mesAtual);
mes.getFormat().getFont().setName(FONTE); mes.getFormat().getFont().setSize(20); mes.getFormat().getFont().setBold(true);
mes.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
mes.getFormat().setVerticalAlignment(ExcelScript.VerticalAlignment.center);

let row = FIRST_DATA_ROW;
let pageEnd = FIRST_DATA_ROW + ROWS_PER_PAGE - 1;

for (const sb of subBlocos) {
if (row + (2 + sb.alunos.length) - 1 > pageEnd) { row = pageEnd + 1; pageEnd = row + ROWS_PER_PAGE - 1; }

// --- Faixa hora (linha row) ---
ws.getRange(`A${row}:P${row}`).getFormat().getFill().setColor(COR_HORA);
const fh = ws.getRange(`A${row}`); fh.setValue(`Hora: ${sb.hora}`);
fh.getFormat().getFont().setName(FONTE); fh.getFormat().getFont().setBold(true); fh.getFormat().getFont().setSize(13);
fh.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.left);
// B:C mesclado = tipo (modalidade)
const bc = ws.getRange(`B${row}:C${row}`); bc.merge(false); bc.setValue(sb.tipo);
bc.getFormat().getFont().setName(FONTE); bc.getFormat().getFont().setBold(true); bc.getFormat().getFont().setSize(12);
bc.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
// Q = Observacoes
const obs = ws.getRange(`Q${row}`); obs.setValue("Observações"); obs.getFormat().getFill().setColor(COR_HORA);
obs.getFormat().getFont().setName(FONTE); obs.getFormat().getFont().setBold(true); obs.getFormat().getFont().setSize(12);
obs.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
ws.getRange(`A${row}:Q${row}`).getFormat().setVerticalAlignment(ExcelScript.VerticalAlignment.center);

// --- Cabecalho de colunas (linha row+1) ---
const hr = ws.getRange(`A${row+1}:Q${row+1}`);
hr.getFormat().getFill().setColor(COR_HEADER); hr.getFormat().getFont().setName(FONTE);
hr.getFormat().getFont().setBold(true); hr.getFormat().getFont().setSize(11);
hr.getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
hr.getFormat().setVerticalAlignment(ExcelScript.VerticalAlignment.center);
ws.getRange(`A${row+1}`).setValue("Aluno(a)");
ws.getRange(`B${row+1}`).setValue("Dias");
ws.getRange(`C${row+1}`).setValue("Livro");
ws.getRange(`D${row+1}`).setValue("Aula");
ws.getRange(`Q${row+1}`).setValue("Professores");

// --- Alunos (linhas row+2 ...) ---
for (let j = 0; j < sb.alunos.length; j++) {
const a = sb.alunos[j]; const rr = row+2+j;
const lr = ws.getRange(`A${rr}:Q${rr}`);
lr.getFormat().getFill().setColor("#FFFFFF");
lr.getFormat().getFont().setName(FONTE); lr.getFormat().getFont().setSize(11);
lr.getFormat().setVerticalAlignment(ExcelScript.VerticalAlignment.center);
ws.getRange(`A${rr}`).setValue(a.nome);
ws.getRange(`A${rr}`).getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.left);
const dias = a.dias.sort((x,y)=>(diaOrd[x]||99)-(diaOrd[y]||99)).map(d=>diaCod[d]||d).join("|");
ws.getRange(`B${rr}`).setValue(dias);
ws.getRange(`C${rr}`).setValue(a.livro);
ws.getRange(`D${rr}`).setValue(a.tipo);
ws.getRange(`B${rr}:D${rr}`).getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
ws.getRange(`Q${rr}`).setValue(a.prof);
ws.getRange(`Q${rr}`).getFormat().setHorizontalAlignment(ExcelScript.HorizontalAlignment.left);
}

// --- Bordas do bloco (A..Q) ---
const bl = ws.getRange(`A${row}:Q${row+1+sb.alunos.length}`).getFormat();
bl.getRangeBorder(ExcelScript.BorderIndex.edgeTop).setStyle(ExcelScript.BorderLineStyle.continuous);
bl.getRangeBorder(ExcelScript.BorderIndex.edgeBottom).setStyle(ExcelScript.BorderLineStyle.continuous);
bl.getRangeBorder(ExcelScript.BorderIndex.edgeLeft).setStyle(ExcelScript.BorderLineStyle.continuous);
bl.getRangeBorder(ExcelScript.BorderIndex.edgeRight).setStyle(ExcelScript.BorderLineStyle.continuous);
bl.getRangeBorder(ExcelScript.BorderIndex.insideHorizontal).setStyle(ExcelScript.BorderLineStyle.continuous);
bl.getRangeBorder(ExcelScript.BorderIndex.insideVertical).setStyle(ExcelScript.BorderLineStyle.continuous);

row += 2 + sb.alunos.length + GAP;
}

// ===== Larguras FIXAS (nunca mudar A,B,C,D,Q) + E:P estreitas =====
// Larguras em PONTOS aproximando o tamanho do template:
ws.getRange("A:A").getFormat().setColumnWidth(220); // Aluno (35)
ws.getRange("B:B").getFormat().setColumnWidth(70); // Dias (11)
ws.getRange("C:C").getFormat().setColumnWidth(100); // Livro (16)
ws.getRange("D:D").getFormat().setColumnWidth(58); // Aula (9)
ws.getRange("E:P").getFormat().setColumnWidth(26); // presenca (estreitas, ajustaveis)
ws.getRange("Q:Q").getFormat().setColumnWidth(100); // Professores (16)

ws.getRange(`A${FIRST_DATA_ROW}:A${row+5}`).getFormat().setRowHeight(ROW_H);

const pl = ws.getPageLayout();
pl.setOrientation(ExcelScript.PageOrientation.landscape);
pl.setPaperSize(ExcelScript.PaperType.a4);
pl.setPrintTitleRows("$1:$4");
pl.setPrintArea(`A1:Q${row+5}`);
pl.setZoom({ horizontalFitToPages: 1, verticalFitToPages: 0 });

resumo.push(`✅ ${aba.sheet}: ${subBlocos.length} sub-bloco(s)`);
}
console.log(resumo.join("\n"));
console.log("✅ Blocos gerados (template fiel).");
}
